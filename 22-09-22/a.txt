#binary search
def bsearch(arr,key,low,high):
    mid=(int)((low+high)/2)
    if(low<high):
        if(arr[mid]==key):
            print("Element found at position",mid)
        elif(key<arr[mid]):
            bsearch(arr,key,low,mid-1)
        elif(key>arr[mid]):
            bsearch(arr,key,mid+1,high)
    else:
        print("Not found")
